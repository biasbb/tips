#### 使用c++的thread类，可以跨平台

1. 使用mutex，同时使用一个全局变量number来计数

   ```c++
   #include<iostream>
   #include <Windows.h>
   #include<thread>
   #include<mutex>
   using namespace std;
   
   
   mutex _mutex;
   int number=0;
   void A()
   {
       while(1)
       {
   
           _mutex.lock();
           if(number%2==1) {
               cout << "A" << endl;
               number++;
               Sleep(200);   //这个是windows的函数，单位是毫秒，开头的S是大写
           }
           _mutex.unlock();
       }
   }
   void B()
   {
       while(1)
       {
           _mutex.lock();
           if(number%2==0)
           {
               cout<<"B"<<endl;
               number++;
               Sleep(200);
           }
           _mutex.unlock();
       }
   }
   int main()
   {
       thread a(A),b(B);
       a.join(),b.join();
   }
   ```

2. 进阶版本，使用条件变量，同时使用unique_lock来托管互斥量mutex，它会在作用域内自动对mutex加锁，同时结束作用域之后会自动解锁，不用手动unlock

   ```c++
   #include<iostream>
   #include <Windows.h>
   #include<thread>
   #include<mutex>
   #include<condition_variable>
   #include<queue>
   using namespace std;
   
   
   mutex data_mutex;
   condition_variable A_var,B_var;
   int number=0;
   queue<int> product;
   int max_size=30;  //定义位队列最大长度
   void A()
   {
       while(1)
       {
           Sleep(1000);
           unique_lock<mutex> lck(data_mutex);
           A_var.wait(lck,[]{return number%2==0;});  //A阻塞，直到number能被2整除，lamada表达式的只有返回为真，才执行下面的，否则阻塞
           cout<<"A"<<endl;
           number++;
           B_var.notify_all();  //唤醒B
       }
   }
   void B()
   {
       while(1)
       {
           Sleep(1000);
           unique_lock<mutex> lck(data_mutex);
           B_var.wait(lck,[]{return number%2==1;});
           cout<<"B"<<endl;
           number++;
           A_var.notify_all();
       }
   }
   int main()
   {
       thread a(A),b(B);
       a.join(),b.join();
   
   }
   ```

3. #### 用互斥量和条件变量实现生产者消费者

   ```c++
   #include<iostream>
   #include <Windows.h>
   #include<thread>
   #include<mutex>
   #include<condition_variable>
   #include<queue>
   using namespace std;
   
   
   mutex data_mutex;
   condition_variable consumer,producer;
   int counter=0;    //记录产品的序号
   queue<int> product;    //保存产品
   int max_size=30;  //队列最多能容纳产品的大小
   void func_consumer()
   {
       while(1)
       {
           Sleep(1000);
           unique_lock<mutex> lck(data_mutex);
           consumer.wait(lck,[]{return product.size()!=0;});
           int cur=product.front();
           product.pop();
           cout<<"消费者消费"<<cur<<"号产品"<<endl;
           producer.notify_all();
       }
   }
   void func_producer()
   {
       while(1)
       {
           Sleep(800);
           unique_lock<mutex> lck(data_mutex);
           producer.wait(lck,[]{return product.size()<max_size;});
           product.push(counter);
           counter++;
           cout<<"生产者生产"<<counter-1<<"号产品"<<endl;
           consumer.notify_all();
       }
   }
   int main()
   {
       thread pro(func_producer);
       thread con(func_consumer);
       pro.join();
       con.join();
   
   }
   ```
