- #### 判断某点在不在三角形里

  二维向量的叉乘的一半即为围城三角形的面积

  ```c++
  #include"iostream"
  #include"cmath"
  #define Precison 0.0001
  using namespace std;
  
  class Point
  {
  public:
      float x;
      float y;
      Point(){}
      Point(int _x,int _y)
      {
          x=_x;
          y=_y;
      }
  
  };
  float GetTriangleSquare(const Point &A,const Point &B,const Point &C)
  {
      Point AB,AC;
      AB.x=B.x-A.x;
      AB.y=B.y-A.y;
      AC.x=C.x-A.x;
      AC.y=C.y-A.y;
      return fabs((AB.x*AC.y-AB.y*AC.x)/2);
  }
  bool IsInTrrangle(const Point &A,const Point &B,const Point &C,const Point &D)
  {
      float SABC, SADB, SBDC, SADC;
      SABC = GetTriangleSquare(A, B, C);
      SADB = GetTriangleSquare(A, D, B);
      SBDC = GetTriangleSquare(B, D, C);
      SADC = GetTriangleSquare(A, D, C);
      float Sum=SADB+SADC+SBDC;
      if(fabs(SABC-Sum)<Precison)
          return true;
      else
          return false;
  }
  int main(){}
  ```
