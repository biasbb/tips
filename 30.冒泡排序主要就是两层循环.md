冒泡排序主要就是两层循环

外循环一共进行len-1次，每次得到一个最大的数放在下面，10个数的话就遍历9次。

内循环是每次遍历比较的次数，为len-i次，i是第i次遍历。10个数的话第一次就比较10-1=9次。

附上一个程序，c艹书145页下面，用数组分别存储学生的姓名和成绩，对成绩进行排序后，按照名次输出名字和对应的分数。

```
#include<iostream>
using namespace std;
int main()
{
    char name[10][10];          
    strcpy(name[0],"Mary");         /将后面的字符串复制到字符数组里，name[0]就代表数组的第0行，即第0                    行整行都存储mary
    strcpy(name[1],"Jam");
    strcpy(name[2],"Jack");
    strcpy(name[3],"Jose");
    strcpy(name[4],"Hery");
    strcpy(name[5],"Mark");
    strcpy(name[6],"Dobbs");
    strcpy(name[7],"Steven");
    strcpy(name[8],"Stanly");
    strcpy(name[9],"John");
    cout<<"请输入对应学生的分数，回车键结束"<<endl;
    int scoretmp,ntmp;
    int score[10][2];    /10行两列的数组，第二列存分数，第一列存name中名字对应的序号
    for(int i=0;i<10;i++)
    {
        score[i][0]=i;
        cout<<name[i]<<endl;
        cin>>score[i][1];
    }
    for(int i=0;i<10;i++)      /用冒泡排序对分数进行排序，同时第一列数字也排了序，对应各自的名字
    {
        for(int j=0;j<9-i;j++)
        {
            if(score[j][1]>score[j+1][1])
            {
                scoretmp=score[j][1];
                ntmp=score[j][0];
                score[j][1]=score[j+1][1];
                score[j][0]=score[j+1][0];
                score[j+1][1]=scoretmp;
                score[j+1][0]=ntmp;
            }
        }
    }
    for(int i=0;i<10;i++)
    {
        cout<<name[score[i][0]]<<endl;
        cout<<score[i][1]<<endl;
    }
    return 0;
}
```
